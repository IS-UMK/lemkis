\section{Introduction to Reinforcement Learning}

Reinforcement learning (RL) is a framework for sequential decision-making under uncertainty. The fundamental probabilistic structure involves an agent interacting with an environment through a stochastic process.

\subsection{The Agent-Environment Interface}

The interaction between agent and environment is characterized by the following probabilistic elements at each discrete time step $t \in \{0, 1, 2, \ldots\}$:

\begin{itemize}
    \item \textbf{State}: $S_t \in \mathcal{S}$ (state space)
    \item \textbf{Action}: $A_t \in \mathcal{A}(s)$ (action space, possibly state-dependent)
    \item \textbf{Reward}: $R_{t+1} \in \mathcal{R} \subseteq \mathbb{R}$ (reward signal)
\end{itemize}

The probabilistic dynamics are governed by the joint probability distribution:
\begin{equation}
    p(s', r \mid s, a) = \mathbb{P}(S_{t+1} = s', R_{t+1} = r \mid S_t = s, A_t = a)
\end{equation}

\section{Markov Decision Processes}

\begin{definition}[Markov Decision Process]
A \textbf{Markov Decision Process (MDP)} is a tuple $\langle \mathcal{S}, \mathcal{A}, p, r, \gamma \rangle$ where:
\begin{itemize}
    \item $\mathcal{S}$ is a finite set of states
    \item $\mathcal{A}$ is a finite set of actions
    \item $p: \mathcal{S} \times \mathcal{A} \times \mathcal{S} \to [0, 1]$ is the state transition probability function, where
    \begin{equation}
        p(s' \mid s, a) = \mathbb{P}(S_{t+1} = s' \mid S_t = s, A_t = a) = \sum_{r \in \mathcal{R}} p(s', r \mid s, a)
    \end{equation}
    \item $r: \mathcal{S} \times \mathcal{A} \to \mathbb{R}$ is the expected reward function
    \begin{equation}
        r(s, a) = \mathbb{E}[R_{t+1} \mid S_t = s, A_t = a] = \sum_{r \in \mathcal{R}} r \sum_{s' \in \mathcal{S}} p(s', r \mid s, a)
    \end{equation}
    \item $\gamma \in [0, 1)$ is the discount factor
\end{itemize}
\end{definition}

\subsection{The Markov Property}

The Markov property states that the future is conditionally independent of the past given the present:
\begin{equation}
    \mathbb{P}(S_{t+1} = s', R_{t+1} = r \mid S_0, A_0, R_1, \ldots, S_t, A_t) = \mathbb{P}(S_{t+1} = s', R_{t+1} = r \mid S_t, A_t)
\end{equation}

This property ensures that the current state $S_t$ contains all relevant information for predicting the future.

\section{Policies and Value Functions}

\subsection{Policies}

\begin{definition}[Policy]
A \textbf{policy} $\pi$ is a probability distribution over actions given states:
\begin{equation}
    \pi(a \mid s) = \mathbb{P}(A_t = a \mid S_t = s)
\end{equation}
where $\sum_{a \in \mathcal{A}} \pi(a \mid s) = 1$ for all $s \in \mathcal{S}$.
\end{definition}

A deterministic policy is a special case where $\pi(a \mid s) \in \{0, 1\}$ for all $s, a$.

\subsection{Return and Value Functions}

\begin{definition}[Return]
The \textbf{return} $G_t$ is the cumulative discounted reward from time $t$:
\begin{equation}
    G_t = \sum_{k=0}^{\infty} \gamma^k R_{t+k+1} = R_{t+1} + \gamma R_{t+2} + \gamma^2 R_{t+3} + \cdots
\end{equation}
\end{definition}

\begin{definition}[State-Value Function]
The \textbf{state-value function} $v_\pi(s)$ under policy $\pi$ is the expected return starting from state $s$:
\begin{equation}
    v_\pi(s) = \mathbb{E}_\pi[G_t \mid S_t = s] = \mathbb{E}_\pi\left[\sum_{k=0}^{\infty} \gamma^k R_{t+k+1} \mid S_t = s\right]
\end{equation}
\end{definition}

\begin{definition}[Action-Value Function]
The \textbf{action-value function} $q_\pi(s, a)$ under policy $\pi$ is the expected return starting from state $s$, taking action $a$:
\begin{equation}
    q_\pi(s, a) = \mathbb{E}_\pi[G_t \mid S_t = s, A_t = a] = \mathbb{E}_\pi\left[\sum_{k=0}^{\infty} \gamma^k R_{t+k+1} \mid S_t = s, A_t = a\right]
\end{equation}
\end{definition}

\subsection{Relationship Between Value Functions}

The state-value and action-value functions are related through:
\begin{equation}
    v_\pi(s) = \sum_{a \in \mathcal{A}} \pi(a \mid s) q_\pi(s, a)
\end{equation}

Conversely:
\begin{equation}
    q_\pi(s, a) = r(s, a) + \gamma \sum_{s' \in \mathcal{S}} p(s' \mid s, a) v_\pi(s')
\end{equation}

\section{Bellman Equations}

\subsection{Bellman Expectation Equations}

\begin{theorem}[Bellman Expectation Equation for $v_\pi$]
The state-value function satisfies the following recursive relationship:
\begin{align}
    v_\pi(s) &= \mathbb{E}_\pi[R_{t+1} + \gamma v_\pi(S_{t+1}) \mid S_t = s] \\
    &= \sum_{a \in \mathcal{A}} \pi(a \mid s) \left[ r(s, a) + \gamma \sum_{s' \in \mathcal{S}} p(s' \mid s, a) v_\pi(s') \right]
\end{align}
\end{theorem}

\begin{theorem}[Bellman Expectation Equation for $q_\pi$]
The action-value function satisfies:
\begin{align}
    q_\pi(s, a) &= \mathbb{E}_\pi[R_{t+1} + \gamma q_\pi(S_{t+1}, A_{t+1}) \mid S_t = s, A_t = a] \\
    &= r(s, a) + \gamma \sum_{s' \in \mathcal{S}} p(s' \mid s, a) \sum_{a' \in \mathcal{A}} \pi(a' \mid s') q_\pi(s', a')
\end{align}
\end{theorem}

\subsection{Bellman Optimality Equations}

\begin{definition}[Optimal Value Functions]
The \textbf{optimal state-value function} is:
\begin{equation}
    v_*(s) = \max_\pi v_\pi(s) = \max_{a \in \mathcal{A}} q_*(s, a)
\end{equation}

The \textbf{optimal action-value function} is:
\begin{equation}
    q_*(s, a) = \max_\pi q_\pi(s, a)
\end{equation}
\end{definition}

\begin{theorem}[Bellman Optimality Equation for $v_*$]
The optimal state-value function satisfies:
\begin{equation}
    v_*(s) = \max_{a \in \mathcal{A}} \left[ r(s, a) + \gamma \sum_{s' \in \mathcal{S}} p(s' \mid s, a) v_*(s') \right]
\end{equation}
\end{theorem}

\begin{theorem}[Bellman Optimality Equation for $q_*$]
The optimal action-value function satisfies:
\begin{equation}
    q_*(s, a) = r(s, a) + \gamma \sum_{s' \in \mathcal{S}} p(s' \mid s, a) \max_{a' \in \mathcal{A}} q_*(s', a')
\end{equation}
\end{theorem}

\begin{definition}[Optimal Policy]
A policy $\pi_*$ is \textbf{optimal} if $v_{\pi_*}(s) = v_*(s)$ for all $s \in \mathcal{S}$. An optimal policy can be derived from $v_*$ or $q_*$:
\begin{equation}
    \pi_*(a \mid s) = \begin{cases}
        1 & \text{if } a \in \arg\max_{a' \in \mathcal{A}} q_*(s, a') \\
        0 & \text{otherwise}
    \end{cases}
\end{equation}
\end{definition}

\section{Dynamic Programming}

\subsection{Policy Evaluation}

\textbf{Goal}: Compute $v_\pi$ for a given policy $\pi$.

\textbf{Iterative Algorithm}: For $k = 0, 1, 2, \ldots$, update:
\begin{equation}
    v_{k+1}(s) = \sum_{a \in \mathcal{A}} \pi(a \mid s) \left[ r(s, a) + \gamma \sum_{s' \in \mathcal{S}} p(s' \mid s, a) v_k(s') \right]
\end{equation}

Under the contraction mapping theorem, $v_k \to v_\pi$ as $k \to \infty$.

\subsection{Policy Improvement}

\begin{theorem}[Policy Improvement Theorem]
Let $\pi$ and $\pi'$ be any pair of deterministic policies such that:
\begin{equation}
    q_\pi(s, \pi'(s)) \geq v_\pi(s) \quad \forall s \in \mathcal{S}
\end{equation}
Then $v_{\pi'}(s) \geq v_\pi(s)$ for all $s \in \mathcal{S}$.
\end{theorem}

The greedy policy with respect to $v_\pi$ is:
\begin{equation}
    \pi'(s) = \arg\max_{a \in \mathcal{A}} \left[ r(s, a) + \gamma \sum_{s' \in \mathcal{S}} p(s' \mid s, a) v_\pi(s') \right]
\end{equation}

\subsection{Policy Iteration}

\textbf{Algorithm}:
\begin{enumerate}
    \item Initialize $\pi_0$ arbitrarily
    \item For $n = 0, 1, 2, \ldots$:
    \begin{enumerate}
        \item \textbf{Policy Evaluation}: Compute $v_{\pi_n}$
        \item \textbf{Policy Improvement}: Set $\pi_{n+1}$ to be greedy w.r.t. $v_{\pi_n}$
        \item If $\pi_{n+1} = \pi_n$, stop and return $\pi_n$ as $\pi_*$
    \end{enumerate}
\end{enumerate}

\subsection{Value Iteration}

Combine policy evaluation and improvement in a single update:
\begin{equation}
    v_{k+1}(s) = \max_{a \in \mathcal{A}} \left[ r(s, a) + \gamma \sum_{s' \in \mathcal{S}} p(s' \mid s, a) v_k(s') \right]
\end{equation}

This is equivalent to turning the Bellman optimality equation into an update rule.

\section{Monte Carlo Methods}

Monte Carlo (MC) methods learn value functions from experience without requiring a model of environment dynamics.

\subsection{Monte Carlo Prediction}

For policy $\pi$, estimate $v_\pi(s)$ by averaging returns following visits to $s$:
\begin{equation}
    v_\pi(s) \approx \frac{1}{N(s)} \sum_{i=1}^{N(s)} G_i(s)
\end{equation}
where $N(s)$ is the number of visits to state $s$, and $G_i(s)$ is the return following the $i$-th visit.

\textbf{First-Visit MC}: Average returns following first visits to $s$ in each episode.

\textbf{Every-Visit MC}: Average returns following all visits to $s$.

By the law of large numbers:
\begin{equation}
    \lim_{N(s) \to \infty} \frac{1}{N(s)} \sum_{i=1}^{N(s)} G_i(s) = \mathbb{E}_\pi[G_t \mid S_t = s] = v_\pi(s)
\end{equation}

\subsection{Incremental Update}

After observing return $G_n$ following the $n$-th visit to state $s$:
\begin{equation}
    v_{n+1}(s) = v_n(s) + \frac{1}{n}(G_n - v_n(s)) = v_n(s) + \alpha_n(G_n - v_n(s))
\end{equation}
where $\alpha_n = 1/n$ is the learning rate.

\section{Temporal Difference Learning}

Temporal Difference (TD) learning combines ideas from Monte Carlo and dynamic programming.

\subsection{TD(0) Prediction}

Update the value function after each step using the observed reward and bootstrapped estimate:
\begin{equation}
    v(S_t) \leftarrow v(S_t) + \alpha [R_{t+1} + \gamma v(S_{t+1}) - v(S_t)]
\end{equation}

The \textbf{TD error} is:
\begin{equation}
    \delta_t = R_{t+1} + \gamma v(S_{t+1}) - v(S_t)
\end{equation}

\subsection{Comparison: MC vs TD}

\textbf{Monte Carlo}:
\begin{equation}
    v(S_t) \leftarrow v(S_t) + \alpha [G_t - v(S_t)]
\end{equation}

\textbf{TD(0)}:
\begin{equation}
    v(S_t) \leftarrow v(S_t) + \alpha [R_{t+1} + \gamma v(S_{t+1}) - v(S_t)]
\end{equation}

TD methods bootstrap (use current estimates), while MC methods use actual returns.

\subsection{SARSA (On-Policy TD Control)}

State-Action-Reward-State-Action updates the action-value function:
\begin{equation}
    q(S_t, A_t) \leftarrow q(S_t, A_t) + \alpha [R_{t+1} + \gamma q(S_{t+1}, A_{t+1}) - q(S_t, A_t)]
\end{equation}

where $A_{t+1} \sim \pi(\cdot \mid S_{t+1})$.

\subsection{Q-Learning (Off-Policy TD Control)}

Q-learning uses the maximum action value for the next state:
\begin{equation}
    q(S_t, A_t) \leftarrow q(S_t, A_t) + \alpha \left[R_{t+1} + \gamma \max_{a'} q(S_{t+1}, a') - q(S_t, A_t)\right]
\end{equation}

The TD error for Q-learning:
\begin{equation}
    \delta_t = R_{t+1} + \gamma \max_{a'} q(S_{t+1}, a') - q(S_t, A_t)
\end{equation}

\section{Exploration vs Exploitation}

\subsection{$\varepsilon$-Greedy Policy}

With probability $\varepsilon$, select a random action; otherwise, select greedily:
\begin{equation}
    \pi(a \mid s) = \begin{cases}
        1 - \varepsilon + \frac{\varepsilon}{|\mathcal{A}(s)|} & \text{if } a = \arg\max_{a'} q(s, a') \\
        \frac{\varepsilon}{|\mathcal{A}(s)|} & \text{otherwise}
    \end{cases}
\end{equation}

\subsection{Softmax Policy (Boltzmann Exploration)}

Select actions according to a Gibbs distribution:
\begin{equation}
    \pi(a \mid s) = \frac{\exp(q(s, a) / \tau)}{\sum_{a' \in \mathcal{A}} \exp(q(s, a') / \tau)}
\end{equation}
where $\tau > 0$ is the temperature parameter.

\section{Function Approximation}

For large state spaces, represent value functions parametrically:
\begin{equation}
    v(s) \approx \hat{v}(s, \mathbf{w})
\end{equation}
where $\mathbf{w} \in \mathbb{R}^d$ is a weight vector.

\subsection{Gradient Descent Update}

Minimize the mean squared value error:
\begin{equation}
    \mathbf{w}_{t+1} = \mathbf{w}_t - \alpha \nabla_{\mathbf{w}} \left[\hat{v}(S_t, \mathbf{w}_t) - v_\pi(S_t)\right]^2
\end{equation}

For TD(0):
\begin{equation}
    \mathbf{w}_{t+1} = \mathbf{w}_t + \alpha [R_{t+1} + \gamma \hat{v}(S_{t+1}, \mathbf{w}_t) - \hat{v}(S_t, \mathbf{w}_t)] \nabla_{\mathbf{w}} \hat{v}(S_t, \mathbf{w}_t)
\end{equation}

\section{Policy Gradient Methods}

Directly parameterize the policy: $\pi(a \mid s, \boldsymbol{\theta})$ where $\boldsymbol{\theta} \in \mathbb{R}^m$.

\subsection{Objective Function}

Maximize the expected return:
\begin{equation}
    J(\boldsymbol{\theta}) = \mathbb{E}_{\pi_{\boldsymbol{\theta}}}[G_0] = \sum_{s \in \mathcal{S}} d^\pi(s) \sum_{a \in \mathcal{A}} \pi(a \mid s, \boldsymbol{\theta}) q_\pi(s, a)
\end{equation}
where $d^\pi(s)$ is the stationary distribution under policy $\pi$.

\subsection{Policy Gradient Theorem}

\begin{theorem}[Policy Gradient Theorem]
\begin{equation}
    \nabla_{\boldsymbol{\theta}} J(\boldsymbol{\theta}) = \mathbb{E}_{\pi_{\boldsymbol{\theta}}} \left[ \nabla_{\boldsymbol{\theta}} \log \pi(A_t \mid S_t, \boldsymbol{\theta}) q_{\pi_{\boldsymbol{\theta}}}(S_t, A_t) \right]
\end{equation}
\end{theorem}

\subsection{REINFORCE Algorithm}

Monte Carlo policy gradient:
\begin{equation}
    \boldsymbol{\theta}_{t+1} = \boldsymbol{\theta}_t + \alpha G_t \nabla_{\boldsymbol{\theta}} \log \pi(A_t \mid S_t, \boldsymbol{\theta}_t)
\end{equation}

\subsection{Actor-Critic Methods}

Use a learned value function (critic) to reduce variance:
\begin{equation}
    \boldsymbol{\theta}_{t+1} = \boldsymbol{\theta}_t + \alpha \delta_t \nabla_{\boldsymbol{\theta}} \log \pi(A_t \mid S_t, \boldsymbol{\theta}_t)
\end{equation}
where $\delta_t = R_{t+1} + \gamma \hat{v}(S_{t+1}, \mathbf{w}) - \hat{v}(S_t, \mathbf{w})$ is the TD error.

\section{Key Probabilistic Concepts in RL}

\subsection{Probability Distributions}

\begin{itemize}
    \item \textbf{Transition Probability}: $p(s' \mid s, a)$ characterizes environmental dynamics
    \item \textbf{Policy}: $\pi(a \mid s)$ defines the agent's behavior
    \item \textbf{State Distribution}: $d_t(s) = \mathbb{P}(S_t = s)$ evolves according to
    \begin{equation}
        d_{t+1}(s') = \sum_{s \in \mathcal{S}} \sum_{a \in \mathcal{A}} d_t(s) \pi(a \mid s) p(s' \mid s, a)
    \end{equation}
\end{itemize}

\subsection{Expectation and Conditional Expectation}

Value functions are conditional expectations:
\begin{align}
    v_\pi(s) &= \mathbb{E}_\pi[G_t \mid S_t = s] \\
    &= \mathbb{E}_\pi \left[ \sum_{k=0}^{\infty} \gamma^k R_{t+k+1} \mid S_t = s \right]
\end{align}

Using the tower property of conditional expectation:
\begin{equation}
    \mathbb{E}_\pi[G_t \mid S_t = s] = \mathbb{E}_\pi[\mathbb{E}_\pi[G_t \mid S_t, A_t] \mid S_t = s]
\end{equation}

\subsection{Law of Total Probability}

Used extensively in deriving Bellman equations:
\begin{equation}
    v_\pi(s) = \sum_{a} \pi(a \mid s) \sum_{s'} \sum_{r} p(s', r \mid s, a) \left[ r + \gamma v_\pi(s') \right]
\end{equation}

\subsection{Convergence and Stochastic Approximation}

TD learning is a form of stochastic approximation. Under Robbins-Monro conditions on the step size $\alpha_t$:
\begin{itemize}
    \item $\sum_{t=0}^{\infty} \alpha_t = \infty$ (guarantees convergence)
    \item $\sum_{t=0}^{\infty} \alpha_t^2 < \infty$ (guarantees variance goes to zero)
\end{itemize}

Typical choice: $\alpha_t = \frac{1}{t+1}$ or constant $\alpha \in (0, 1]$ for non-stationary environments.

\section{Summary}

Reinforcement learning provides a probabilistic framework for sequential decision-making:

\begin{itemize}
    \item \textbf{Environment}: Modeled as an MDP with transition probabilities $p(s' \mid s, a)$
    \item \textbf{Agent}: Characterized by a policy $\pi(a \mid s)$
    \item \textbf{Value Functions}: Expected returns $v_\pi(s)$ and $q_\pi(s, a)$
    \item \textbf{Bellman Equations}: Recursive decomposition of expectations
    \item \textbf{Learning Algorithms}: Methods to estimate value functions and improve policies using observed data
\end{itemize}

The probabilistic nature of RL is fundamental: randomness in the environment, stochasticity in policies, and statistical estimation from samples all play crucial roles in the theory and practice of reinforcement learning.
